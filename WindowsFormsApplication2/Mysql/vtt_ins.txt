CREATE DEFINER=`root`@`%` PROCEDURE `vtt_ins`(
aId varchar(20),
aName varchar(255), 
aBrand varchar(100), 
aDescription varchar(500),
aGroup varchar(200), 
aRootGroup varchar(200),
aAvailableQuantity decimal(15,2),
aTransitQuantity decimal(15,2),
aPrice decimal(15,2),
aPriceRetail decimal(15,2),
aWidth decimal(15,2),
aHeight decimal(15,2), 
aDepth decimal(15,2),
aWeight decimal(15,2) ,
aPhotoUrl longtext, 
aOriginalNumber varchar(45), 
aVendor varchar(45),
aCompatibility LONGTEXT, 
aBarcode varchar(45), 
aMainOfficeQuantity decimal(15,2), 
aNumberInPackage decimal(15,2), 
aColorName varchar(45), 
aTransitDate date, 
aItemLifeTime varchar(45),
aNameAlias varchar(45)
)
BEGIN
DECLARE parId varchar(20);
	set parId = (select Id from ItemDto where Id = aId);
    IF parId is NULL THEN
		insert into ItemDto (
Id,
Name,
Brand,
Description,
GroupName,
RootGroup,
AvailableQuantity,
TransitQuantity,
Price,
PriceRetail,
Width,
Height,
Depth,
Weight,
PhotoUrl,
OriginalNumber,
Vendor,
Compatibility,
Barcode,
MainOfficeQuantity,
NumberInPackage,
ColorName,
TransitDate,
ItemLifeTime,
NameAlias) values (
aId,
aName,
aBrand,
aDescription,
aGroup,
aRootGroup,
aAvailableQuantity,
aTransitQuantity,
aPrice,
aPriceRetail,
aWidth,
aHeight,
aDepth,
aWeight,
aPhotoUrl,
aOriginalNumber,
aVendor,
aCompatibility,
aBarcode,
aMainOfficeQuantity,
aNumberInPackage,
aColorName,
aTransitDate,
aItemLifeTime,
aNameAlias
);
    ELSE
		update ItemDto SET
Name = aName,
Brand = aBrand,
Description = aDescription,
GroupName = aGroup,
RootGroup = aRootGroup,
AvailableQuantity = aAvailableQuantity,
TransitQuantity = aTransitQuantity,
Price = aPrice,
PriceRetail = aPriceRetail,
Width = aWidth,
Height = aHeight,
Depth = aDepth,
Weight = aWeight,
PhotoUrl = aPhotoUrl,
OriginalNumber = aOriginalNumber,
Vendor = aVendor,
Compatibility = aCompatibility,
Barcode = aBarcode,
MainOfficeQuantity = aMainOfficeQuantity,
NumberInPackage = aNumberInPackage,
ColorName = aColorName,
TransitDate = aTransitDate,
ItemLifeTime = aItemLifeTime,
NameAlias = aNameAlias
where Id = aId;
    END IF;
END
